#include<stdio.h>
#include<stdlib.h>



int main()
{
    int n,q;
    scanf("%d %d",&n,&q);
    int *a=(int*)malloc(n*sizeof(int));
    int i=0,flag;
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    int min=a[0],max=a[n-1];
    for(i=1;i<n;i++)
    {
        if(a[i]>max)
            max=a[i];
        if(a[i]<min)
            min=a[i];
    }
    free(a);
    while(q--)
    {
        scanf("%d",&flag);
        if(flag>=min&&flag<=max)
            printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
